84 etiss::int8 SimpleMemSystem::load_elf(const char *elf_file)
85 {
86  ELFIO::elfio reader;
87 
88  if (!reader.load(elf_file))
89  {
90  etiss::log(etiss::ERROR, "ELF reader could not process file");
91  return -1;
92  }
93  // set architecture automatically
94  if (reader.get_machine() == EM_RISCV)
95  {
96  if ((reader.get_class() == ELFCLASS64)) {
97  etiss::cfg().set<std::string>("arch.cpu", "RISCV64"); // RISCV and OR1K work as well
98  } else if ((reader.get_class() == ELFCLASS32)) {
99  etiss::cfg().set<std::string>("arch.cpu", "RISCV");
100  // add conditions
101  } else {
102  etiss::log(etiss::ERROR, "System architecture is neither 64 nor 32 bit");
103  return -1;
104  }
105  }
106  else if (reader.get_machine() == EM_OPENRISC)
107  {
108  if ((reader.get_class() == ELFCLASS32))
109  {
110  etiss::cfg().set<std::string>("arch.cpu", "OR1K");
111  }
112  else if ((reader.get_class() == ELFCLASS64))
113  {
114  etiss::log(etiss::ERROR, "OR1k 64 is not supported");
115  return -1;
116  }
117  }
118  else
119  {
120  etiss::log(etiss::ERROR, "Architecture in ELF is not supported");
121  return -1;
122  }
123 
124  for (auto &seg : reader.segments)
125  {
126  std::unique_ptr<MemSegment> mseg;
127  etiss::uint64 start_addr = seg->get_physical_address();
128  etiss::uint64 size = seg->get_memory_size();
129  size_t file_size = seg->get_file_size();
130  MemSegment::access_t mode = (seg->get_flags() & PF_W) ? MemSegment::WRITE : MemSegment::READ;
131  std::stringstream sname;
132  sname << seg->get_index() << " - " << std::hex << std::setfill('0') << (mode == MemSegment::WRITE ? "W" : "R")
133  << "[0x" << std::setw(sizeof(etiss::uint64) * 2) << start_addr + size - 1 << " - "
134  << "0x" << std::setw(sizeof(etiss::uint64) * 2) << start_addr << "]";
135 
136  bool newseg_valid = true;
137  for (const auto &mseg_it : msegs_)
138  {
139  if ((start_addr >= mseg_it->start_addr_) && (start_addr <= mseg_it->end_addr_))
140  {
141  std::stringstream msg;
142  msg << "Segment " << sname.str() << "already occupied by another segment\n";
143  etiss::log(etiss::WARNING, msg.str().c_str());
144  newseg_valid = false;
145  break;
146  }
147  }
148  if (newseg_valid)
149  {
150  if ((start_addr >= rom_start_) && (start_addr < (rom_start_ + rom_size_)))
151  {
152  if (size > rom_size_)
153  {
154  etiss::log(etiss::FATALERROR, "ELF segment does not fit in predetermined ROM size");
155  }
156  mseg = std::make_unique<MemSegment>(start_addr, size, mode, sname.str(),
157  rom_mem_.data() + (start_addr - rom_start_));
158  }
159  else if ((start_addr >= ram_start_) && (start_addr < (ram_start_ + ram_size_)))
160  {
161  if (size > ram_size_)
162  {
163  etiss::log(etiss::FATALERROR, "ELF segment does not fit in predetermined RAM size");
164  }
165  mseg = std::make_unique<MemSegment>(start_addr, size, mode, sname.str(),
166  ram_mem_.data() + (start_addr - ram_start_));
167  }
168  else if (rom_size_ == 0 && ram_size_ == 0)
169  { // system memory is dynamically allocated during ELF load (self managed by each memory segment)
170  mseg = std::make_unique<MemSegment>(start_addr, size, mode, sname.str());
171  }
172  else
173  {
174  break;
175  }
176  if (mseg)
177  {
178  add_memsegment(std::move(mseg), seg->get_data(), file_size);
179  }
180  }
181  }
182 
183  // read start or rather program boot address from ELF
184  start_addr_ = reader.get_entry();
185 
186  return 0;
187 }

278 {

284  for (i_seg = 0; i_seg < n_segs; ++i_seg)

311  return RETURNCODE::DBUS_READ_ERROR;
312  }
313  }
